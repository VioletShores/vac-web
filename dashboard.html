function buildTrustGraph(user, contacts, vouches) { const svgEl = document.getElementById('trustGraphSvg'); if (!svgEl) return; const svg = d3.select('#trustGraphSvg'); svg.selectAll('*').remove(); const width = svgEl.clientWidth || 700; const height = 320; svg.attr('viewBox', `0 0 ${width} ${height}`); const nodes = [{ id: email, name: user.name || 'You', score: user.trust_score || 0, type: 'self', radius: 36, verifications: user.total_verifications || 0 }]; const links = []; const seen = new Set([email]); contacts.forEach(c => { const isMutual = c.i_vouched && c.they_vouched; nodes.push({ id: c.email, name: c.name || c.email.split('@')[0], score: c.trust_score || 0, type: isMutual ? 'mutual' : 'voucher', radius: 26 }); seen.add(c.email); links.push({ source: email, target: c.email, mutual: isMutual }); }); (vouches.vouches_received || []).forEach(v => { if (!seen.has(v.email)) { nodes.push({ id: v.email, name: v.person || v.email.split('@')[0], score: 0, type: 'voucher', radius: 22 }); seen.add(v.email); links.push({ source: email, target: v.email, mutual: false }); } }); // Outside node nodes.push({ id: '_out', name: '?', score: 0, type: 'outside', radius: 18 }); links.push({ source: email, target: '_out', mutual: false }); const colors = { self: '#2DD4BF', mutual: '#10B981', voucher: '#7C3AED', outside: '#374151' }; const simulation = d3.forceSimulation(nodes) .force('link', d3.forceLink(links).id(d => d.id).distance(130)) .force('charge', d3.forceManyBody().strength(-350)) .force('center', d3.forceCenter(width / 2, height / 2)) .force('collision', d3.forceCollide().radius(d => d.radius + 30)) .force('y', d3.forceY(height / 2).strength(0.08)); const g = svg.append('g'); // Links g.append('g').selectAll('line').data(links).join('line') .attr('stroke', d => d.mutual ? colors.mutual : '#1E2230') .attr('stroke-width', d => d.mutual ? 2.5 : 1.5) .attr('stroke-dasharray', d => d.mutual ? 'none' : '6,4') .attr('opacity', 0.6) .attr('class', 'graph-link'); // Node groups const node = g.append('g').selectAll('g').data(nodes).join('g') .attr('cursor', 'pointer') .call(d3.drag() .on('start', (e, d) => { if (!e.active) simulation.alphaTarget(0.3).restart(); d.fx = d.x; d.fy = d.y; }) .on('drag', (e, d) => { d.fx = e.x; d.fy = e.y; }) .on('end', (e, d) => { if (!e.active) simulation.alphaTarget(0); d.fx = null; d.fy = null; }) ); // Click handler — show detail panel node.on('click', (e, d) => { if (d.type === 'outside') return; const panel = document.getElementById('nodeDetail'); const avatar = document.getElementById('detailAvatar'); const c = colors[d.type]; avatar.textContent = d.name.charAt(0).toUpperCase(); avatar.style.background = c + '20'; avatar.style.border = '2px solid ' + c; avatar.style.color = c; document.getElementById('detailName').textContent = d.name; document.getElementById('detailEmail').textContent = d.id; document.getElementById('detailScore').textContent = d.score > 0 ? d.score.toFixed(2) : '—'; const statusEl = document.getElementById('detailStatus'); statusEl.textContent = d.type === 'self' ? 'YOU · ROOT' : d.type === 'mutual' ? 'MUTUAL VOUCH' : 'VOUCHED FOR YOU'; statusEl.style.color = c; panel.style.display = 'block'; // Highlight this node node.selectAll('circle').attr('opacity', 0.3); d3.select(e.currentTarget).selectAll('circle').attr('opacity', 1); setTimeout(() => node.selectAll('circle').attr('opacity', 1), 2000); }); // Glow for self node.filter(d => d.type === 'self').append('circle') .attr('r', d => d.radius + 8).attr('fill', 'none') .attr('stroke', colors.self).attr('stroke-width', 1.5).attr('opacity', 0.2); // Score arc (visual trust ring) node.filter(d => d.score > 0 && d.type !== 'outside').each(function(d) { const arc = d3.arc().innerRadius(d.radius + 2).outerRadius(d.radius + 5) .startAngle(0).endAngle(d.score * 2 * Math.PI); d3.select(this).append('path').attr('d', arc) .attr('fill', colors[d.type]).attr('opacity', 0.5); }); // Main circles node.append('circle').attr('r', d => d.radius) .attr('fill', d => (colors[d.type] || '#374151') + '12') .attr('stroke', d => colors[d.type] || '#374151') .attr('stroke-width', d => d.type === 'self' ? 2.5 : d.type === 'outside' ? 1 : 2) .attr('stroke-dasharray', d => d.type === 'outside' ? '4,3' : 'none'); // Initial letter (centered in circle) node.append('text').attr('text-anchor', 'middle').attr('dy', '0.38em') .attr('fill', d => colors[d.type] || '#374151') .attr('font-size', d => d.type === 'self' ? '18px' : d.type === 'outside' ? '14px' : '15px') .attr('font-weight', '700').attr('font-family', 'var(--mono)') .text(d => d.type === 'outside' ? '?' : d.name.charAt(0).toUpperCase()); // Name label BELOW circle node.filter(d => d.type !== 'outside').append('text') .attr('text-anchor', 'middle').attr('y', d => d.radius + 16) .attr('fill', '#E5E7EB').attr('font-size', '11px').attr('font-weight', '600') .attr('font-family', 'var(--sans)') .text(d => { const n = d.name; return n.length > 14 ? n.substring(0, 14) + '…' : n; }); // Outside label node.filter(d => d.type === 'outside').append('text') .attr('text-anchor', 'middle').attr('y', d => d.radius + 14) .attr('fill', '#6B7280').attr('font-size', '9px').attr('font-weight', '600') .attr('font-family', 'var(--mono)').attr('letter-spacing', '0.5px') .text('OUTSIDE'); // Tick simulation.on('tick', () => { g.selectAll('.graph-link') .attr('x1', d => d.source.x).attr('y1', d => d.source.y) .attr('x2', d => d.target.x).attr('y2', d => d.target.y); node.attr('transform', d => `translate(${Math.max(40, Math.min(width-40, d.x))},${Math.max(40, Math.min(height-40, d.y))})`); }); } loadDashboard();
